## Overview

Form-related UI lives under `shared/components/ui/`. Low-level controls include `Input`, `Textarea`, `Select`, `Checkbox`, `Switch`, `RadioGroup`, and `NativeSelect`. For labels use `Label`. For react-hook-form integration use `Form*` from `form.tsx`. For layout-oriented field groups use `Field*` from `field.tsx`. For grouped inputs with addons use `InputGroup*` from `input-group.tsx`.

> **Note:** `Select`, `Checkbox`, `Switch`, `RadioGroup`, `Form`, `Field`, and `InputGroup` parts are client components (`"use client"`). Use them from client components or pass serialized props from Server Components.

## Basic inputs

### Input

```tsx
import { Input } from "@/shared/components/ui/input";

<Input type="email" name="email" placeholder="you@example.com" />
```

Set `aria-invalid` when validation fails so focus/error styles apply.

### Textarea

```tsx
import { Textarea } from "@/shared/components/ui/textarea";

<Textarea name="bio" rows={4} placeholder="Tell us about yourself" />
```

### Label

```tsx
import { Label } from "@/shared/components/ui/label";

<Label htmlFor="email">Email</Label>
```

## Select

Radix-based combobox-style select (`Select` from `radix-ui`):

```tsx
"use client";

import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/shared/components/ui/select";

<Select defaultValue="one">
  <SelectTrigger className="w-[180px]">
    <SelectValue placeholder="Choose" />
  </SelectTrigger>
  <SelectContent>
    <SelectItem value="one">One</SelectItem>
    <SelectItem value="two">Two</SelectItem>
  </SelectContent>
</Select>
```

`SelectTrigger` accepts `size?: "sm" | "default"`.

### Native select

Styled `<select>` for simple cases:

```tsx
import {
  NativeSelect,
  NativeSelectOption,
} from "@/shared/components/ui/native-select";

<NativeSelect name="role" defaultValue="user" size="sm">
  <NativeSelectOption value="user">User</NativeSelectOption>
  <NativeSelectOption value="admin">Admin</NativeSelectOption>
</NativeSelect>
```

## React Hook Form (`Form`)

Wrap with `Form` (`FormProvider`), then `FormField` + `FormItem` + `FormLabel` + `FormControl` + `FormMessage`:

```tsx
"use client";

import { useForm } from "react-hook-form";
import {
  Form,
  FormControl,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
} from "@/shared/components/ui/form";
import { Input } from "@/shared/components/ui/input";

const form = useForm({ defaultValues: { title: "" } });

function onSubmit(values: { title: string }) {
  console.log(values);
}

<Form {...form}>
  <form onSubmit={form.handleSubmit(onSubmit)}>
    <FormField
      control={form.control}
      name="title"
      render={({ field }) => (
        <FormItem>
          <FormLabel>Title</FormLabel>
          <FormControl>
            <Input {...field} />
          </FormControl>
          <FormMessage />
        </FormItem>
      )}
    />
  </form>
</Form>
```

## Field primitives (`Field`)

Use for accessible field layout without tying to react-hook-form: `FieldSet`, `FieldLegend`, `FieldGroup`, `Field`, `FieldLabel`, `FieldContent`, `FieldDescription`, `FieldError`, `FieldSeparator`, etc. `Field` supports `orientation`: `"vertical" | "horizontal" | "responsive"`.

## Input group

Combine addons, text, and controls in one bordered control:

```tsx
"use client";

import {
  InputGroup,
  InputGroupAddon,
  InputGroupInput,
} from "@/shared/components/ui/input-group";

<InputGroup>
  <InputGroupAddon>https://</InputGroupAddon>
  <InputGroupInput placeholder="yoursite.com" />
</InputGroup>
```

Also available: `InputGroupButton`, `InputGroupText`, `InputGroupTextarea`. Set `data-align` on addons: `inline-start`, `inline-end`, `block-start`, `block-end`.
